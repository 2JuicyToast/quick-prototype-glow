        // Generate Significantly Bigger and Brighter Sparkling Stars
        function createStars() {
            const container = document.getElementById('star-container');
            const starCount = Math.floor(Math.random() * 21) + 40; // Slightly fewer stars to accommodate larger size
            const colors = ['#d0bcff', '#adc6ff', '#4fdbc8']; // Brand primary, secondary, tertiary

            for (let i = 0; i < starCount; i++) {
                const star = document.createElement('div');
                star.className = 'star';
                
                // Significantly increased base size (4px to 10px instead of 1-3px)
                const size = Math.random() * 6 + 4; 
                const x = Math.random() * 100;
                const y = Math.random() * 100;
                const color = colors[Math.floor(Math.random() * colors.length)];
                const duration = Math.random() * 2.5 + 1.5; // Faster twinkle for "sparkle" feel
                const delay = Math.random() * 5;

                star.style.width = `${size}px`;
                star.style.height = `${size}px`;
                star.style.left = `${x}%`;
                star.style.top = `${y}%`;
                star.style.backgroundColor = color;
                // Amplified glow intensity
                star.style.boxShadow = `0 0 ${size * 3}px ${color}, 0 0 ${size * 5}px ${color}88`;
                star.style.setProperty('--duration', `${duration}s`);
                star.style.animationDelay = `${delay}s`;

                container.appendChild(star);
            }
        }
        createStars();

        // Micro-interactions for hover effects
        document.querySelectorAll('.glass-card').forEach(card => {
            card.addEventListener('mousemove', e => {
                const rect = card.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                card.style.setProperty('--mouse-x', `${x}px`);
                card.style.setProperty('--mouse-y', `${y}px`);
            });
        });

        // Simple scroll effect for TopNavBar
        window.addEventListener('scroll', () => {
            const nav = document.querySelector('nav');
            if (window.scrollY > 20) {
                nav.classList.add('bg-surface/95', 'shadow-lg');
                nav.classList.remove('bg-surface/80');
            } else {
                nav.classList.add('bg-surface/80');
                nav.classList.remove('bg-surface/95', 'shadow-lg');
            }
        });
    </script>
