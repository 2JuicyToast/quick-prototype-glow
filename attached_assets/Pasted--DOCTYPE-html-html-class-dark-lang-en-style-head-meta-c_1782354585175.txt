<script>
    // Subtle Mouse parallax for the background star field and nebula
    document.addEventListener('mousemove', (e) => {
        const x = e.clientX / window.innerWidth;
        const y = e.clientY / window.innerHeight;
        
        const starField = document.querySelector('.star-field');
        const intensity = 15;
        
        starField.style.transform = `translate(${x * intensity}px, ${y * intensity}px)`;
        
        const glows = document.querySelectorAll('.nebula-glow');
        glows.forEach((glow, index) => {
            const move = (index + 1) * 30;
            glow.style.transform = `translate(${(x - 0.5) * move}px, ${(y - 0.5) * move}px)`;
        });
    });
</script>
